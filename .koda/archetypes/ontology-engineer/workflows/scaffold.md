### Step 3: Generate Type Definitions

The workflow will:

1. **Inspect Snowflake schema** to discover tables and columns

4. **Define type hierarchies** based on business logic

**Example Output** (`src/ontology/types.py`):

```python
"""
Type Definitions for Address Matching Ontology

Generated by: Ontology Engineer Archetype
Date: 2025-10-09
"""

import relationalai as rai
from src.config import Config

def create_model(config: Config) -> rai.Model:
    """Create RAI model with all type definitions"""
    model_name = config.get_model_full_name()
    model = rai.Model(model_name)
    
    # Raw types from Snowflake
    RawUSPSCity = model.Type('RawUSPSCity',
        source=config.get_data_source('usps_cities'))
    
    # Entity types
    City = model.Type('City')
    USPSCity = model.Type('USPSCity')
    UnrecognizedCity = model.Type('UnrecognizedCity')
    City.extend(USPSCity, UnrecognizedCity)
    
    return model
```

4. **Data correction** (fix common typos, standardize formats)

```python
"""
Data Cleansing Rules

These rules standardize address components before matching.
"""

from src.ontology.types import model, Address
from relationalai.std.strings import upper, trim

# RULE: Normalize city names to uppercase
# Rationale: Enables case-insensitive matching
with model.rule():
    address = Address()
    address.set(
        cleansed_city = upper(trim(address.original_city))
    )

# RULE: Expand directional abbreviations
# Example: "N WILKESBORO" → "NORTH WILKESBORO"
with model.rule():
    address = Address()
    starts_with(address.original_city, 'N ')
    address.set(
        cleansed_city = replace(address.original_city, 'N ', 'NORTH ')
    )
```

```python
"""
Entity Matching Rules

These rules identify when two entities represent the same real-world object.
"""

from src.ontology.types import model, Address, AddressMatch

# RULE: Exact match on standardized address
# When all components match exactly after cleansing
with model.rule():
    addr1 = Address()
    addr2 = Address()
    addr1.cleansed_street == addr2.cleansed_street
    addr1.cleansed_city == addr2.cleansed_city
    addr1.cleansed_state == addr2.cleansed_state
    AddressMatch.add(address1=addr1, address2=addr2, match_type='exact')
```

```python
"""
Common Query Patterns

Reusable query functions for the ontology.
"""

from src.ontology.types import model

def get_all_matches(match_type: str = None):
    """
    Retrieve all address matches, optionally filtered by type.
    
    Args:
        match_type: Optional filter ('exact', 'fuzzy', 'phonetic')
        
    Returns:
        DataFrame with matched address pairs
    """
    with model.query() as select:
        match = AddressMatch()
        if match_type:
            match.match_type == match_type
        return select(
            match.address1.original_street,
            match.address2.original_street,
            match.match_type
        )
```

```python
"""
Deploy RAI Ontology Model

Usage:
    python CDO-AIFC/reference/scripts/deploy_ontology.py --config CDO-AIFC/reference/config/dev.yaml --environment dev
"""

import argparse
from src.config import load_config
from src.ontology.types import create_model
import relationalai as rai

def deploy_model(config_path: str, environment: str):
    """Deploy ontology model to RAI"""
    config = load_config(config_path)
    model = create_model(config)
    
    # Connect to RAI
    rai_client = rai.Client(profile=config.get('relationalai.profile'))
    
    # Deploy model
    rai_client.deploy_model(model)
    print(f"Model deployed to {environment}")

if __name__ == '__main__':
    parser = argparse.ArgumentParser()
    parser.add_argument('--config', required=True)
    parser.add_argument('--environment', required=True)
    args = parser.parse_args()
    
    deploy_model(args.config, args.environment)
```
