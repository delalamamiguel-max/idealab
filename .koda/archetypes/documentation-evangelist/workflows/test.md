**Test Script Structure** (`tests/test_documentation.sh` or `tests/test_documentation.py`):

```bash
#!/bin/bash
# Documentation Test Harness
# Generated by Documentation Evangelist

set -euo pipefail

DOCS_DIR="${1:-.}"
EXIT_CODE=0

echo "🧪 Running Documentation Tests..."
echo "=================================="

# Test 1: Markdown Linting
test_markdown_lint() {
  echo "📝 Test 1: Markdown Linting"
  if markdownlint "$DOCS_DIR/**/*.md" --config .markdownlint.json; then
    echo "✅ PASS: Markdown lint"
  else
    echo "❌ FAIL: Markdown lint errors found"
    EXIT_CODE=1
  fi
}

# Test 2: Mermaid Diagram Validation
test_mermaid_diagrams() {
  echo "📊 Test 2: Mermaid Diagram Validation"
  local failed=0
  for file in $(find "$DOCS_DIR" -name "*.md"); do
    if grep -q "```mermaid" "$file"; then
      if ! mmdc --input "$file" --validate; then
        echo "❌ FAIL: Invalid Mermaid in $file"
        failed=1
      fi
    fi
  done
  if [ $failed -eq 0 ]; then
    echo "✅ PASS: All Mermaid diagrams valid"
  else
    EXIT_CODE=1
  fi
}

# Test 3: Link Validation
test_links() {
  echo "🔗 Test 3: Link Validation"
  if markdown-link-check "$DOCS_DIR/**/*.md" --config .markdown-link-check.json; then
    echo "✅ PASS: All links valid"
  else
    echo "❌ FAIL: Broken links found"
    EXIT_CODE=1
  fi
}

# Test 4: Code Example Validation
test_code_examples() {
  echo "💻 Test 4: Code Example Validation"
  # Extract and validate code blocks
  python3 tests/validate_code_blocks.py "$DOCS_DIR"
  if [ $? -eq 0 ]; then
    echo "✅ PASS: All code examples valid"
  else
    echo "❌ FAIL: Invalid code examples found"
    EXIT_CODE=1
  fi
}

# Test 5: Metadata Validation
test_metadata() {
  echo "📋 Test 5: Metadata Validation"
  local failed=0
  for file in $(find "$DOCS_DIR" -name "*.md"); do
    if ! grep -q "^---" "$file"; then
      echo "❌ FAIL: Missing metadata in $file"
      failed=1
    fi
  done
  if [ $failed -eq 0 ]; then
    echo "✅ PASS: All files have metadata"
  else
    EXIT_CODE=1
  fi
}

# Test 6: Required Sections
test_required_sections() {
  echo "📑 Test 6: Required Sections"
  python3 tests/check_required_sections.py "$DOCS_DIR"
  if [ $? -eq 0 ]; then
    echo "✅ PASS: All required sections present"
  else
    echo "❌ FAIL: Missing required sections"
    EXIT_CODE=1
  fi
}

# Test 7: Line Length
test_line_length() {
  echo "📏 Test 7: Line Length (max 100 chars)"
  local failed=0
  for file in $(find "$DOCS_DIR" -name "*.md"); do
    if awk 'length > 100 {print FILENAME":"NR":"$0; exit 1}' "$file"; then
      :
    else
      echo "❌ FAIL: Line length violation in $file"
      failed=1
    fi
  done
  if [ $failed -eq 0 ]; then
    echo "✅ PASS: All lines within limit"
  else
    EXIT_CODE=1
  fi
}

# Run all tests
test_markdown_lint
test_mermaid_diagrams
test_links
test_code_examples
test_metadata
test_required_sections
test_line_length

echo "=================================="
if [ $EXIT_CODE -eq 0 ]; then
  echo "✅ All documentation tests passed!"
else
  echo "❌ Some documentation tests failed"
fi

exit $EXIT_CODE
```

**Supporting Test Scripts**:

1. **`tests/validate_code_blocks.py`**: Extracts code blocks, validates syntax for each language
2. **`tests/check_required_sections.py`**: Verifies presence of required sections
3. **`.markdownlint.json`**: Markdown linting configuration
4. **`.markdown-link-check.json`**: Link checking configuration

**CI Integration** (`.github/CDO-AIFC/reference/workflows/docs-test.yml` or similar):
```yaml
name: Documentation Tests

on: [push, pull_request]

jobs:
  test-docs:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v2
      - name: Install tools
        run: |
          npm install -g markdownlint-cli
          npm install -g @mermaid-js/mermaid-cli
          npm install -g markdown-link-check
      - name: Run documentation tests
        run: bash tests/test_documentation.sh
```
